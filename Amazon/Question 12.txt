class Solution{
    public:
    string colName (long long int n)
    {
        // your code here
        string ans;
        while(n>0){
            long long int rem=n%26;
            if(rem==0){
                ans.push_back('Z');
                n=(n-1)/26;
            }
            else{
            ans.push_back(rem+64);
            n=n/26;
            }
        }
        reverse(ans.begin(),ans.end());
        return ans;
    }
};

	{
		long long int n; cin >> n;
		Solution ob;
		cout << ob.colName (n) << '\n';
	}
}
